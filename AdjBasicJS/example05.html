Example 05

Drop Down Lists

What month were you born?



This example shows how to get the value of a dropdown list. In this case I use an onchange event to populate the div.

<form action="#">
	<h2>Drop Down Lists</h2>
	<h4>What month were you born?</h4>
	<select id="DDBirthMonth" onchange="DisplayMonth();">
		<option value="Init">Please select the month you were born</option>
		<option value="January">January</option>
		<option value="February">February</option>
		<option value="March">March</option>
		<option value="April">April</option>
		<option value="May">May</option>
		<option value="June">June</option>
		<option value="July">July</option>
		<option value="August">August</option>
		<option value="September">September</option>
		<option value="October">October</option>
		<option value="November">November</option>
		<option value="December">December</option>
	</select>
	<br />
	<br />
	<div id="DivMessage" style="font-weight:bold;"></div>
</form>

<script type="text/javascript">

function DisplayMonth() {
    var ddMonth = document.getElementById('DDBirthMonth');
    var ddText = ddMonth.options[ddMonth.selectedIndex].text;
    var ddValue = ddMonth.options[ddMonth.selectedIndex].value;
    var myDiv = document.getElementById('DivMessage');
    myDiv.innerHTML = "Your birth month is " + ddText;
    ddMonth.selectedIndex = 0;
}

</script>
