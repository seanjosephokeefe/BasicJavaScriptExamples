Example 03

What's your favorite food?

Pizza    Hamburgers    Twinkies    All of the Above    I don't eat food, I'm a robot.

This example shows how to get the value of a radio button in a radio button group.

<h3>Example 03</h3>
<form action="#">
	<h2>What's your favorite food?</h2>
	<p>
		<input id="RBPizza" name="RGFood" type="radio" value ="Pizza"/>Pizza   
		<input id="RBHamburgers" name="RGFood" type="radio" value="Hamburgers" />Hamburgers   
		<input id="RBTwinkies" name="RGFood" type="radio" value="Twinkies" />Twinkies   
		<input id="RBAll" name="RGFood" type="radio" checked="checked"  
		    value="All of the Above"/>All of the Above   
		<input id="RBRobot" name="RGFood" type="radio" value="You're a stinkin' robot."/>I 
		don't eat food, I'm a robot.
	</p>
	<div id="DivMessage"></div>
	<input id="BtnFoodType" type="button" value="Get Food Type" onclick="GetFoodChoice();" />
	<input id="BtnReset" type="reset" value="Clear" onclick="ClearDiv('DivMessage');" />
</form>

<script type="text/javascript">

function GetFoodChoice() {
var myDiv = document.getElementById('DivMessage');
var message;
if (document.getElementById('RBPizza').checked)
	message = document.getElementById('RBPizza').value;
else if (document.getElementById('RBHamburgers').checked)
	message = document.getElementById('RBHamburgers').value;
else if (document.getElementById('RBTwinkies').checked)
	message = document.getElementById('RBTwinkies').value;
else if (document.getElementById('RBAll').checked)
	message = document.getElementById('RBAll').value;
else if (document.getElementById('RBRobot').checked)
	message = document.getElementById('RBRobot').value;
myDiv.innerHTML = message;
}

function ClearDiv(thisDiv) {
	var myDiv = document.getElementById(thisDiv);
	myDiv.innerHTML = "";
}

</script>
