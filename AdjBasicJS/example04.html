This example shows how to get the values of checkboxes.

<form action="#">
	<h3>Checking Checkboxes</h3>
	<h4>What sports do you like?</h4>
	<p><input id="CBFootball" type="checkbox" value="Football" />Football</p>
	<p><input id="CBBaseball" type="checkbox" value="Baseball" />Baseball</p>
	<p><input id="CBSoccer" type="checkbox" value="Soccer" />Soccer</p>
	<p><input id="CBNASCAR" type="checkbox" value="NASCAR" />NASCAR</p>
	<p><input id="CBHockey" type="checkbox" value="Hockey" />Hockey</p>
	<div id="DivMessage"></div>
	<br />
	<input id="BtnSportTypes" type="button" value="Show me my choices" 
		onclick="DisplaySports();" />
	<input id="BtnReset" type="reset" value="Clear" onclick="ClearDiv('DivMessage');" />
</form>

<script type="text/javascript">

function DisplaySports() {
    var myDiv = document.getElementById('DivMessage');
    var message = "You like the following sports: ";
    message = ReturnMessage("CBFootball", message);
    message = ReturnMessage("CBBaseball", message);
    message = ReturnMessage("CBSoccer", message);
    message = ReturnMessage("CBNASCAR", message);
    message = ReturnMessage("CBHockey", message);
    if (message == "You like the following sports: ")
        message += "None selected";
    myDiv.innerHTML = message;
}

function ReturnMessage(whichControl, message) {
    if (document.getElementById(whichControl).checked == true) {
        if (message != "You like the following sports: ") {
            message += ",";
        }
        return message += document.getElementById(whichControl).value;
    }
    return message;
}

function ClearDiv(thisDiv) {
	var myDiv = document.getElementById(thisDiv);
	myDiv.innerHTML = "";
}

</script>
