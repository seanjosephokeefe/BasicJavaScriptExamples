<script type="text/javascript">

function ManipulateControls() {
    var message = "Your answers:";
    message += "<br />";
    var myDiv = document.getElementById('DivMessage');
    var holiday = document.getElementById("TBHoliday").value;
    if (holiday.length == 0)
        message += "You didn't answer your favorite holiday so I put Halloween. ";
    else
        message += "You answered " + holiday + " for your favorite holiday but it should have been 
		Halloween. ";
    message += "<br />";
    var ddDay = document.getElementById('DDWkEndDay').options[document.getElementById('DDWkEndDay')
	.selectedIndex].text;
    message += "Day of the week, ";
    if (ddDay == "Friday")
        message += " Right, Friday starts the weekend so it rocks.";
    else if (ddDay == "Saturday")
        message += " Saturday's ok. But Friday is the best.";
    else if (ddDay == "Sunday")
        message += " Sunday is the end of the weekend.";
    message += "<br />";
    var pet;
    if (document.getElementById('RBDog').checked == true)
        message += " You're right, dogs are the coolest. ";
    else if (document.getElementById('RBCat').checked == true)
        message += " You think cats are cool? Wow!!! ";
    else if (document.getElementById('RBTurtle').checked == true)
        message += " You like turtles. ";
    else if (document.getElementById('RBNone').checked == true)
        message += " Why no pets? Pets rock, especially dogs. ";
    message += "<br />";
    myDiv.innerHTML = message + " Your fave color should be clear. <br />Don't worry, I fixed your 
	selections for you.<br /><br />";
    document.getElementById('DDWkEndDay').selectedIndex = 0;
    document.getElementById('RBDog').checked = true;
    document.getElementById("TBHoliday").value = "Halloween, of course. Awesome movies";
    document.getElementById("CBYellow").checked = false;
    document.getElementById("CBBlue").checked = false;
    document.getElementById("CBRed").checked = false;
}

function ClearDiv(thisDiv) {
    var myDiv = document.getElementById(thisDiv);
    myDiv.innerHTML = "";
}

</script>
